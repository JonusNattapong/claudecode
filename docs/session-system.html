Session Start

src/utils/sessionStart.ts
async function startSession() {
  // 1. Load settings
  const settings = await loadSettings()

  // 2. Load memories
  const memories = await loadMemoryPrompt()

  // 3. Load plugins
  await loadPluginHooks()

  // 4. Build system prompt
  const systemPrompt = buildSystemPrompt(settings, memories)

  // 5. Start message loop
  await messageLoop(systemPrompt)
}

Session Resume

src/utils/sessionRestore.ts
async function resumeSession(sessionId: string) {
  // 1. Load transcript
  const transcript = await loadTranscript(sessionId)

  // 2. Restore state
  const state = await loadSessionState(sessionId)

  // 3. Reconstruct context
  const messages = rebuildMessages(transcript, state)

  // 4. Continue session
  return messages
}

Key Learnings

  1. JSONL transcript — Messages stored as JSONL for efficient appending
  2. State separate from transcript — Metadata stored separately
  3. Resume capability — Full state reconstruction possible
  4. Session URLs — Bridge mode uses shareable URLs
  5. Activity tracking — Last active timestamp for cleanup
